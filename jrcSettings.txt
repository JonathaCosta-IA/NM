# --------------------------------------------------------
# REMOVER AMBIENTE CONDA / MAMBA
# --------------------------------------------------------
echo "Removendo ambientes conda..."

if command -v conda &> /dev/null; then
    conda deactivate 2>/dev/null

    # remover ambiente científico
    conda env remove -n sci -y 2>/dev/null

    # remover base (Miniforge)
    rm -rf ~/miniforge3
    rm -rf ~/Miniforge3
fi

# limpar bashrc (conda + variáveis)
sed -i '/conda initialize/,+10d' ~/.bashrc
sed -i '/OMP_NUM_THREADS/d' ~/.bashrc
sed -i '/OPENBLAS_NUM_THREADS/d' ~/.bashrc

# --------------------------------------------------------
# REMOVER PACOTES PYTHON GLOBAIS
# --------------------------------------------------------
echo "Removendo pacotes pip..."
pip uninstall -y pynvim 2>/dev/null

# --------------------------------------------------------
# REMOVER PACOTES APT INSTALADOS
# --------------------------------------------------------
echo "Removendo pacotes do sistema..."

sudo apt remove --purge -y \
build-essential cmake git curl wget htop neofetch \
pkg-config software-properties-common unzip zip \
libopenblas-dev liblapack-dev gfortran \
python3-dev python3-pip \
mesa-utils libgl1-mesa-dri libglx-mesa0 \
neovim nodejs npm tmux ripgrep fd-find fzf \
texlive-latex-base texlive-latex-extra \
texlive-fonts-recommended texlive-science \
latexmk

sudo apt autoremove -y
sudo apt clean

# --------------------------------------------------------
# REMOVER VS CODE (SNAP)
# --------------------------------------------------------
echo "Removendo VS Code..."
sudo snap remove code 2>/dev/null

# --------------------------------------------------------
# REMOVER SWAP CRIADO
# --------------------------------------------------------
echo "Removendo swap..."

sudo swapoff /swapfile 2>/dev/null
sudo rm -f /swapfile

# --------------------------------------------------------
# REMOVER ALIAS
# --------------------------------------------------------
sed -i '/alias fdfind=fd/d' ~/.bashrc

# --------------------------------------------------------
# FINALIZAÇÃO
# --------------------------------------------------------
echo "Limpeza concluída."
echo "Reinicie o sistema para garantir estado limpo."
